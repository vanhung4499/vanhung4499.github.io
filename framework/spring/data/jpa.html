    @Column(unique = true)

    private Integer age;

    private String address;

    private String email;

        this.age = age;
        this.address = address;
        this.email = email;

    @Override

    }

    @Override

            return true;
        }

        if (!(o instanceof User)) {
            return false;
        }

        User user = (User) o;

        if (id != null && id.equals(user.id)) {
            return true;
        }
