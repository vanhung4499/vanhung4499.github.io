class Solution:
    def findTargetSumWays(self, nums: List[int], target: int) -> int:
        size = len(nums)
        table = dict()

        def dfs(i, cur_sum):
            if i == size:

                else:
                    return 0

    def my_tribonacci(self, n: int, memo: List[int]) -> int:
